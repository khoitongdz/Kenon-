-- Auto Farm Logic
local isAutoFarming = false

local function startAutoFarm()
    while isAutoFarming do
        -- Logic to automatically farm monsters goes here
        print("Farming monsters...")
        -- Example of XP Gain and Level Up
        local xp = 100
        local level = 1
        xp = xp + 10
        if xp >= level * 100 then
            level = level + 1
            print("Level Up! Now at Level " .. level)
        end
        wait(1) -- Delay to prevent too fast farming
    end
end

-- Button Click Handlers
autoFarmButton.MouseButton1Click:Connect(function()
    isAutoFarming = true
    print("Auto Farm Started!")
    startAutoFarm()
end)
