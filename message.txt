-- Fast Attack Toggle Button
local fastAttackEnabled = false
local xp = 0
local level = 1

local player = game.Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()

-- Function to find the nearest enemy
local function findNearestEnemy()
    local nearestEnemy = nil
    local shortestDistance = math.huge
    for _, enemy in pairs(game.Workspace.Enemies:GetChildren()) do
        if enemy:FindFirstChild("HumanoidRootPart") and enemy:FindFirstChild("Humanoid") then
            local distance = (character.HumanoidRootPart.Position - enemy.HumanoidRootPart.Position).Magnitude
            if distance < shortestDistance then
                shortestDistance = distance
                nearestEnemy = enemy
            end
        end
    end
    return nearestEnemy
end

-- Function to perform a fast attack
local function fastAttack()
    while fastAttackEnabled do
        wait(0.1)  -- Attack every 0.1 seconds for fast attacks
        local target = findNearestEnemy()
        if target then
            -- Move to the enemy's position
            character.HumanoidRootPart.CFrame = target.HumanoidRootPart.CFrame
            -- Perform attack (customize the attack method for your game)
            game.ReplicatedStorage:WaitForChild("RemoteEvent"):FireServer("Attack", target)
            xp = xp + 10  -- Gain XP for attacking (example)
            if xp >= level * 100 then
                level = level + 1
                xp = 0
            end
        end
    end
end

-- Function to start auto attack (with fast attack enabled)
local function startAutoAttackFunc()
    fastAttackEnabled = true
    fastAttack()  -- Start fast attack loop
end

-- Function to stop auto attack
local function stopAutoAttackFunc()
    fastAttackEnabled = false
end

-- Button Click Events
startAutoAttack.MouseButton1Click:Connect(function()
    startAutoAttackFunc()
end)

stopAutoAttack.MouseButton1Click:Connect(function()
    stopAutoAttackFunc()
end)
