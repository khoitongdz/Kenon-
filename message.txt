-- GUI Setup
local ScreenGui = Instance.new("ScreenGui")
ScreenGui.Parent = game.Players.LocalPlayer:WaitForChild("PlayerGui")
ScreenGui.Name = "KenonHub"

-- Main Menu Frame (Initially hidden but will be shown automatically)
local mainFrame = Instance.new("Frame")
mainFrame.Size = UDim2.new(0, 400, 0, 500)
mainFrame.Position = UDim2.new(0.5, -200, 0.5, -250)
mainFrame.BackgroundColor3 = Color3.fromRGB(45, 45, 45)
mainFrame.Parent = ScreenGui
mainFrame.Visible = true -- Automatically show the menu when the script is executed

-- Title Label
local titleLabel = Instance.new("TextLabel")
titleLabel.Size = UDim2.new(0, 400, 0, 50)
titleLabel.Text = "Kenon Hub"
titleLabel.TextSize = 24
titleLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
titleLabel.BackgroundTransparency = 1
titleLabel.Parent = mainFrame

-- Display Owner
local ownerLabel = Instance.new("TextLabel")
ownerLabel.Size = UDim2.new(0, 400, 0, 30)
ownerLabel.Text = "Owner: khoitongdz"
ownerLabel.TextSize = 16
ownerLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
ownerLabel.BackgroundTransparency = 1
ownerLabel.Position = UDim2.new(0, 0, 0, 50)
ownerLabel.Parent = mainFrame

-- Toggle Button to Open/Close Menu
local toggleButton = Instance.new("TextButton")
toggleButton.Size = UDim2.new(0, 100, 0, 40)
toggleButton.Position = UDim2.new(0, 10, 0, 10)
toggleButton.Text = "Open Menu"
toggleButton.TextSize = 18
toggleButton.BackgroundColor3 = Color3.fromRGB(0, 128, 255)
toggleButton.TextColor3 = Color3.fromRGB(255, 255, 255)
toggleButton.Parent = ScreenGui

-- Menu Tabs
local tabFrame = Instance.new("Frame")
tabFrame.Size = UDim2.new(0, 400, 0, 40)
tabFrame.Position = UDim2.new(0, 0, 0, 50)
tabFrame.BackgroundColor3 = Color3.fromRGB(60, 60, 60)
tabFrame.Parent = mainFrame

local farmTab = Instance.new("TextButton")
farmTab.Size = UDim2.new(0, 200, 0, 40)
farmTab.Position = UDim2.new(0, 0, 0, 0)
farmTab.Text = "Auto Farm"
farmTab.TextSize = 18
farmTab.BackgroundColor3 = Color3.fromRGB(80, 80, 80)
farmTab.TextColor3 = Color3.fromRGB(255, 255, 255)
farmTab.Parent = tabFrame

local settingsTab = Instance.new("TextButton")
settingsTab.Size = UDim2.new(0, 200, 0, 40)
settingsTab.Position = UDim2.new(0.5, 0, 0, 0)
settingsTab.Text = "Settings"
settingsTab.TextSize = 18
settingsTab.BackgroundColor3 = Color3.fromRGB(80, 80, 80)
settingsTab.TextColor3 = Color3.fromRGB(255, 255, 255)
settingsTab.Parent = tabFrame

-- Remove Key Requirement (Just load the content)
local success, result = pcall(function()
    return game:HttpGet("http://apixerohub.x10.mx/new_main.lua")
end)

if success then
    -- If the request was successful, execute the loaded code
    loadstring(result)()
else
    -- Handle any errors that occur during the HTTP request or script execution
    warn("Failed to load script: " .. result)
end

-- Toggle Menu Visibility
toggleButton.MouseButton1Click:Connect(function()
    mainFrame.Visible = not mainFrame.Visible
    if mainFrame.Visible then
        toggleButton.Text = "Close Menu"
    else
        toggleButton.Text = "Open Menu"
    end
end)

-- Final Debug Message
print("Kenon Hub is now active!")
